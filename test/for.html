var mvvm = new MVVM({
            el: '#app',
            data: {
                companies: [{
                    name: 'microsoft',
                    location: 'US'
                }, {
                    name: 'tencent',
                    location: 'China'
                }],
                message: 'hello MVVM',
            },
            methods: {
                changeList: function() {
                    var randomStr = function(times) {
                        var ret = []
                        for(var i = 0; i < times; i++) {
                            ret.push(String.fromCharCode(97 + Math.floor(26 * Math.random(0, 1))));
                        }
                        return ret.join('');
                    }
                    this.companies.push({name:randomStr(8), location: 'US'});
                }
            }
        });
